option(ENABLE_ZOLTAN "Enable Zoltan itergace [ON|OFF]" OFF)
message(STATUS "ENABLE_ZOLTAN: " ${ENABLE_ZOLTAN})

set(CMAKE_MODULE_PATH
  ${CMAKE_MODULE_PATH}
  "${CMAKE_CURRENT_SOURCE_DIR/../cmake/}")

if (ENABLE_ZOLTAN)
  find_package(Zoltan MODULE REQUIRED)
endif()
if (ENABLE_KOKKOS)
  find_package(Kokkos MODULE REQUIRED)
endif()
include_directories(
  ../agi
  ../zoltan
  ../interfaces
  ${ZOLTAN_INCLUDE_DIRS}
  ${PARMETIS_INCLUDE_DIRS}
  ${KOKKOS_INCLUDE_DIRS}
  )

set(ALL_LIBS
  agi
  interfaces
  agi_zoltan
  SCOREC::core
  ${ZOLTAN_LIBRARIES}
  ${PARMETIS_LIBRARIES}
  ${KOKKOS_LIBRARIES}
  )

function(make_exe exename srcname)
  add_executable(${exename} ${srcname})
  target_link_libraries(${exename} ${ALL_LIBS})
endfunction(make_exe)

make_exe(buildMeshGraph buildMeshGraph.cpp)
